## ListBox

This Doc will gives you a brief description of the ListBox
component.

Follow below code snippet to use ListBox component.

```tsx
import { ListBox } from '@nx-repo/design-system';

return (
  <ThemeContextProvider theme={defaultTheme}>
    <div
      style={{
        display: 'flex',
        minWidth: '200px',
        border: '1px solid lightgray',
        maxHeight: 300,
      }}
    >
      <ListBox flexGrow={1} aria-labelledby="label" items={flatOptions}>
        {(item) => <Item key={item.name}>{item.name}</Item>}
      </ListBox>
    </div>
  </ThemeContextProvider>
);
```

**ListBox Props**

```tsx
interface ListBoxProps<T> extends SpectrumListBoxProps<T> {
  customOption?: (props: any) => JSX.Element;
  listLayoutOptions?: ListLayoutOptions<T>;
}
```
